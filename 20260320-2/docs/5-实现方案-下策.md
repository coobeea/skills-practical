### 3.1 核心诊断引擎

**diagnostic_engine.py** (主程序,约500行)

```python
#!/usr/bin/env python3
"""
CRS信息披露诊断核心引擎
负责: 整个诊断流程的协调和控制
"""

import json
from pathlib import Path
from question_controller import QuestionController
from validator import Validator
from session_manager import SessionManager
from risk_assessor import RiskAssessor
from report_generator import ReportGenerator

class DiagnosticEngine:
    def __init__(self):
        self.question_controller = QuestionController()
        self.validator = Validator()
        self.session_manager = SessionManager()
        self.risk_assessor = RiskAssessor()
        self.report_generator = ReportGenerator()
        
        self.current_session = None
    
    def start_diagnostic(self, account_name):
        """启动新的诊断会话"""
        self.current_session = self.session_manager.create_session(account_name)
        first_question = self.question_controller.get_question("1")
        
        return {
            "status": "started",
            "session_id": self.current_session['session_id'],
            "question": first_question
        }
    
    def process_answer(self, session_id, question_id, user_answer):
        """
        处理用户答案
        
        Returns:
            - 如果有下一个问题: {"status": "continue", "next_question": {...}}
            - 如果分类完成: {"status": "category_complete", "next_category": "..."}
            - 如果全部完成: {"status": "complete", "report": {...}}
        """
        # 1. 加载会话
        session = self.session_manager.load_session(session_id)
        
        # 2. 校验答案
        validation_result = self.validator.validate(question_id, user_answer)
        if not validation_result['valid']:
            return {
                "status": "error",
                "error": validation_result['error'],
                "suggestion": validation_result['suggestion']
            }
        
        # 3. 保存答案
        self.session_manager.save_answer(
            session_id,
            question_id,
            user_answer,
            validation_result['structured_answer']
        )
        
        # 4. 判定下一个问题
        next_step = self.question_controller.get_next(
            question_id,
            validation_result['structured_answer']
        )
        
        if next_step['status'] == 'continue':
            return {
                "status": "continue",
                "next_question": next_step['question']
            }
        
        elif next_step['status'] == 'category_complete':
            # 检查是否还有其他分类需要诊断
            remaining_categories = self._get_remaining_categories(session)
            if remaining_categories:
                next_category = remaining_categories[0]
                next_question = self.question_controller.get_first_question_of_category(next_category)
                return {
                    "status": "category_complete",
                    "completed_category": next_step['category'],
                    "next_category": next_category,
                    "next_question": next_question
                }
            else:
                # 所有分类完成,生成报告
                return self._generate_report(session_id)
        
        elif next_step['status'] == 'complete':
            # 所有问题完成,生成报告
            return self._generate_report(session_id)
    
    def _get_remaining_categories(self, session):
        """获取剩余需要诊断的资产类型"""
        # 从Q1的答案中获取用户选择的资产类型
        q1_answer = self.session_manager.get_answer(session['session_id'], "1")
        selected_assets = q1_answer['structured_answer']
        
        # 获取已完成的分类
        answered_questions = self.session_manager.get_answered_questions(session['session_id'])
        completed_categories = set()
        for q_id in answered_questions:
            question = self.question_controller.get_question(q_id)
            completed_categories.add(question['category'])
        
        # 计算剩余分类
        category_mapping = {
            "境外银行存款": "境外银行存款",
            "境外证券投资": "境外证券投资",
            "离岸公司": "离岸公司",
            "境外保单": "境外保单",
            "离岸信托": "离岸信托"
        }
        
        remaining = []
        for asset in selected_assets:
            category = category_mapping[asset]
            if category not in completed_categories:
                remaining.append(category)
        
        return remaining
    
    def _generate_report(self, session_id):
        """生成诊断报告"""
        session = self.session_manager.load_session(session_id)
        
        # 1. 风险评估
        risk_assessment = self.risk_assessor.assess(session)
        
        # 2. 生成报告
        report = self.report_generator.generate(session, risk_assessment)
        
        # 3. 保存报告
        self.session_manager.save_report(session_id, report)
        
        return {
            "status": "complete",
            "report": report
        }
    
    def resume_session(self, session_id):
        """恢复中断的会话"""
        session = self.session_manager.load_session(session_id)
        
        # 获取最后回答的问题
        answered_questions = self.session_manager.get_answered_questions(session_id)
        if not answered_questions:
            # 没有回答任何问题,从Q1开始
            first_question = self.question_controller.get_question("1")
            return {
                "status": "resumed",
                "session_id": session_id,
                "progress": 0,
                "next_question": first_question
            }
        
        last_question_id = answered_questions[-1]
        last_answer = self.session_manager.get_answer(session_id, last_question_id)
        
        # 获取下一个问题
        next_step = self.question_controller.get_next(
            last_question_id,
            last_answer['structured_answer']
        )
        
        return {
            "status": "resumed",
            "session_id": session_id,
            "progress": len(answered_questions),
            "last_question": last_question_id,
            "next_question": next_step['question'] if next_step['status'] == 'continue' else None
        }

# CLI接口
if __name__ == "__main__":
    import sys
    
    engine = DiagnosticEngine()
    
    command = sys.argv[1] if len(sys.argv) > 1 else None
    
    if command == "start":
        # python diagnostic_engine.py start "张先生_2025"
        account_name = sys.argv[2]
        result = engine.start_diagnostic(account_name)
    
    elif command == "answer":
        # python diagnostic_engine.py answer <session_id> <question_id> <answer>
        session_id = sys.argv[2]
        question_id = sys.argv[3]
        answer = sys.argv[4]
        result = engine.process_answer(session_id, question_id, answer)
    
    elif command == "resume":
        # python diagnostic_engine.py resume <session_id>
        session_id = sys.argv[2]
        result = engine.resume_session(session_id)
    
    else:
        result = {"error": "用法: python diagnostic_engine.py <start|answer|resume> [参数]"}
    
    print(json.dumps(result, ensure_ascii=False, indent=2))
```

**question_controller.py** (约200行)

```python
#!/usr/bin/env python3
"""
问题流程控制器
负责: 问题的加载、跳转逻辑判定
"""

import json
from pathlib import Path

class QuestionController:
    def __init__(self):
        self.questions = self._load_questions()
        self.questions_dict = {q['id']: q for q in self.questions}
    
    def _load_questions(self):
        """从JSON文件加载问题清单"""
        config_file = Path(__file__).parent.parent / 'config' / 'questions.json'
        with open(config_file, 'r', encoding='utf-8') as f:
            data = json.load(f)
            return data['questions']
    
    def get_question(self, question_id):
        """获取指定问题"""
        return self.questions_dict.get(question_id)
    
    def get_first_question_of_category(self, category):
        """获取指定分类的第一个问题"""
        for q in self.questions:
            if q['category'] == category:
                return q
        return None
    
    def get_next(self, current_id, user_answer):
        """
        根据当前问题和用户答案,确定下一步
        
        Returns:
            {"status": "continue", "question": {...}}
            {"status": "category_complete", "category": "..."}
            {"status": "complete"}
        """
        current_question = self.get_question(current_id)
        if not current_question:
            return {"status": "error", "error": f"问题 {current_id} 不存在"}
        
        # 判定下一个问题ID
        next_id = None
        
        if 'jump_logic' in current_question:
            # 有条件跳转
            jump_logic = current_question['jump_logic']
            
            # 如果答案是字符串,直接查找
            if isinstance(user_answer, str):
                next_id = jump_logic.get(user_answer)
            
            # 如果答案是列表(多选),取第一个
            elif isinstance(user_answer, list) and user_answer:
                next_id = jump_logic.get(user_answer[0])
        
        # 如果jump_logic没有匹配,使用默认next
        if not next_id:
            next_id = current_question.get('next')
        
        if not next_id:
            # 没有下一个问题,当前分类结束
            return {
                "status": "category_complete",
                "category": current_question['category']
            }
        
        next_question = self.get_question(next_id)
        if not next_question:
            return {"status": "error", "error": f"下一个问题 {next_id} 不存在"}
        
        return {
            "status": "continue",
            "question": next_question
        }
```

#!/usr/bin/env python3
"""
答案校验器
负责: 校验答案格式、调用地区匹配、货币换算等子模块
"""

import json
from jurisdiction_matcher import JurisdictionMatcher
from currency_converter import CurrencyConverter

class Validator:
    def __init__(self):
        self.jurisdiction_matcher = JurisdictionMatcher()
        self.currency_converter = CurrencyConverter()
    
    def validate(self, question_id, user_answer):
        """
        校验答案
        
        Returns:
            {
                "valid": True/False,
                "error": "错误描述"(如果invalid),
                "suggestion": "建议"(如果invalid),
                "structured_answer": {...}(如果valid)
            }
        """
        # 根据问题类型进行不同的校验
        question = self._get_question(question_id)
        question_type = question['type']
        
        if question_type == 'multiple_choice':
            return self._validate_multiple_choice(question, user_answer)
        
        elif question_type == 'single_choice':
            return self._validate_single_choice(question, user_answer)
        
        elif question_type == 'text_input':
            return self._validate_text_input(question, user_answer)
        
        else:
            return {
                "valid": False,
                "error": f"未知问题类型: {question_type}"
            }
    
    def _validate_multiple_choice(self, question, user_answer):
        """校验多选题"""
        options = question['options']
        
        # 如果用户输入是字符串,尝试解析
        if isinstance(user_answer, str):
            user_answer = [ans.strip() for ans in user_answer.split(',')]
        
        # 检查每个选项是否有效
        invalid_options = [ans for ans in user_answer if ans not in options]
        if invalid_options:
            return {
                "valid": False,
                "error": f"无效选项: {', '.join(invalid_options)}",
                "suggestion": f"请从以下选项中选择: {', '.join(options)}"
            }
        
        return {
            "valid": True,
            "structured_answer": user_answer
        }
    
    def _validate_single_choice(self, question, user_answer):
        """校验单选题"""
        options = question['options']
        
        if user_answer not in options:
            return {
                "valid": False,
                "error": f"无效选项: {user_answer}",
                "suggestion": f"请从以下选项中选择: {', '.join(options)}"
            }
        
        return {
            "valid": True,
            "structured_answer": user_answer
        }
    
    def _validate_text_input(self, question, user_answer):
        """校验文本输入"""
        extraction_type = question.get('extraction')
        
        if extraction_type == 'country':
            # 地区名称,需要匹配
            match_result = self.jurisdiction_matcher.match(user_answer)
            if 'error' in match_result:
                return {
                    "valid": False,
                    "error": match_result['error'],
                    "suggestion": match_result['suggestion']
                }
            
            return {
                "valid": True,
                "structured_answer": match_result
            }
        
        elif extraction_type == 'amount':
            # 金额,需要换算
            convert_result = self.currency_converter.convert(user_answer)
            if 'error' in convert_result:
                return {
                    "valid": False,
                    "error": convert_result['error'],
                    "suggestion": convert_result['suggestion']
                }
            
            return {
                "valid": True,
                "structured_answer": convert_result
            }
        
        else:
            # 普通文本,不做特殊处理
            return {
                "valid": True,
                "structured_answer": user_answer
            }
    
    def _get_question(self, question_id):
        """获取问题信息(从question_controller)"""
        from question_controller import QuestionController
        controller = QuestionController()
        return controller.get_question(question_id)
```

#!/usr/bin/env python3
"""
风险评估器
负责: 根据用户答案评估CRS信息披露风险
"""

import json
from pathlib import Path

class RiskAssessor:
    def __init__(self):
        self.risk_rules = self._load_risk_rules()
    
    def _load_risk_rules(self):
        """加载风险评估规则"""
        config_file = Path(__file__).parent.parent / 'config' / 'risk_rules.json'
        with open(config_file, 'r', encoding='utf-8') as f:
            return json.load(f)
    
    def assess(self, session):
        """
        评估会话的整体风险
        
        Returns:
            {
                "overall_risk_level": "高风险/中风险/低风险",
                "assets": [
                    {
                        "asset_type": "境外银行存款",
                        "risk_level": "高风险",
                        "risk_factors": ["CRS参与国", "中国税收居民", "高净值账户"],
                        "disclosure_obligation": "有披露义务"
                    },
                    ...
                ],
                "risk_points": [
                    {
                        "title": "高净值境外银行存款已被CRS交换",
                        "description": "...",
                        "impact": "...",
                        "urgency": "紧急"
                    },
                    ...
                ]
            }
        """
        answers = session['answers']
        
        # 获取用户选择的资产类型(Q1答案)
        q1_answer = self._get_answer(answers, "1")
        selected_assets = q1_answer['structured_answer']
        
        # 对每个资产进行风险评估
        asset_assessments = []
        for asset_type in selected_assets:
            assessment = self._assess_asset(asset_type, answers)
            asset_assessments.append(assessment)
        
        # 确定总体风险等级(取最高)
        risk_levels = [a['risk_level'] for a in asset_assessments]
        if "高风险" in risk_levels:
            overall_risk = "高风险"
        elif "中风险" in risk_levels:
            overall_risk = "中风险"
        else:
            overall_risk = "低风险"
        
        # 识别关键风险点
        risk_points = self._identify_risk_points(asset_assessments, answers)
        
        return {
            "overall_risk_level": overall_risk,
            "assets": asset_assessments,
            "risk_points": risk_points
        }
    
    def _assess_asset(self, asset_type, answers):
        """评估单个资产的风险"""
        if asset_type == "境外银行存款":
            return self._assess_bank_account(answers)
        elif asset_type == "境外证券投资":
            return self._assess_securities(answers)
        elif asset_type == "离岸公司":
            return self._assess_offshore_company(answers)
        elif asset_type == "境外保单":
            return self._assess_insurance(answers)
        elif asset_type == "离岸信托":
            return self._assess_trust(answers)
        else:
            return {"error": f"未知资产类型: {asset_type}"}
    
    def _assess_bank_account(self, answers):
        """评估境外银行存款风险"""
        # 获取相关答案
        country = self._get_answer(answers, "1-1")['structured_answer']
        holder_type = self._get_answer(answers, "1-2")['structured_answer']
        is_china_resident = self._get_answer(answers, "1-2-1")['structured_answer']
        balance_range = self._get_answer(answers, "1-3-2")['structured_answer']
        is_special = self._get_answer(answers, "1-4")['structured_answer']
        
        # 风险因素
        risk_factors = []
        
        # 判定1: CRS参与国
        if not country.get('is_crs'):
            return {
                "asset_type": "境外银行存款",
                "risk_level": "低风险",
                "risk_factors": ["非CRS参与国"],
                "disclosure_obligation": "无披露义务"
            }
        risk_factors.append("CRS参与国")
        
        # 判定2: 税收居民身份
        if is_china_resident != "是中国税收居民":
            return {
                "asset_type": "境外银行存款",
                "risk_level": "低风险",
                "risk_factors": risk_factors + ["非中国税收居民"],
                "disclosure_obligation": "无披露义务"
            }
        risk_factors.append("中国税收居民")
        
        # 判定3: 账户余额
        if balance_range == "超过250万美元":
            risk_factors.append("高净值账户(>250万美元)")
            risk_level = "高风险"
        elif balance_range == "100万至250万美元":
            risk_factors.append("中等净值账户(100-250万美元)")
            risk_level = "中风险"
        else:
            risk_factors.append("普通账户(<100万美元)")
            risk_level = "中风险"
        
        # 判定4: 豁免账户
        if is_special == "是特殊类型账户":
            risk_factors.append("可能属于豁免账户")
            # 降低风险等级
            if risk_level == "高风险":
                risk_level = "中风险"
        
        return {
            "asset_type": "境外银行存款",
            "risk_level": risk_level,
            "risk_factors": risk_factors,
            "disclosure_obligation": "有披露义务",
            "basic_info": {
                "country": country['name_cn'],
                "holder_type": holder_type,
                "is_china_resident": is_china_resident,
                "balance_range": balance_range
            }
        }
    
    # 其他资产评估方法 ...
    # _assess_securities, _assess_offshore_company, _assess_insurance, _assess_trust
    
    def _identify_risk_points(self, asset_assessments, answers):
        """识别关键风险点"""
        risk_points = []
        
        for assessment in asset_assessments:
            if assessment['risk_level'] == "高风险":
                # 生成高风险点描述
                if assessment['asset_type'] == "境外银行存款":
                    risk_points.append({
                        "title": "高净值境外银行存款已被CRS交换",
                        "description": f"您的{assessment['basic_info']['country']}银行账户余额{assessment['basic_info']['balance_range']},属于高净值账户。",
                        "impact": "账户信息已通过CRS交换至中国税务机关",
                        "urgency": "紧急"
                    })
                # ... 其他资产类型
        
        # 按紧急程度排序
        urgency_order = {"紧急": 0, "中等": 1, "低": 2}
        risk_points.sort(key=lambda x: urgency_order[x['urgency']])
        
        return risk_points
    
    def _get_answer(self, answers, question_id):
        """从答案列表中获取指定问题的答案"""
        for answer in answers:
            if answer['question_id'] == question_id:
                return answer
        return None
```

#!/usr/bin/env python3
"""
报告生成器
负责: 根据会话数据和风险评估结果生成诊断报告
"""

import json
from pathlib import Path
from datetime import datetime

class ReportGenerator:
    def __init__(self):
        self.report_config = self._load_report_config()
    
    def _load_report_config(self):
        """加载报告配置"""
        config_file = Path(__file__).parent.parent / 'config' / 'report_config.json'
        with open(config_file, 'r', encoding='utf-8') as f:
            return json.load(f)
    
    def generate(self, session, risk_assessment):
        """
        生成诊断报告
        
        Returns:
            完整的Markdown格式报告(字符串)
        """
        report_sections = []
        
        # 第一章: 诊断概要
        report_sections.append(self._generate_overview(session, risk_assessment))
        
        # 第二章: 资产总览
        report_sections.append(self._generate_asset_overview(risk_assessment))
        
        # 第三章: 各资产信息披露义务判定
        report_sections.append(self._generate_asset_analysis(session, risk_assessment))
        
        # 第四章: CRS信息交换风险评估
        report_sections.append(self._generate_risk_assessment(risk_assessment))
        
        # 第五章: 合规建议
        report_sections.append(self._generate_compliance_advice(risk_assessment))
        
        # 第六章: 参考依据
        report_sections.append(self._generate_references())
        
        # 拼接完整报告
        report = "\n\n".join(report_sections)
        
        return report
    
    def _generate_overview(self, session, risk_assessment):
        """生成诊断概要"""
        account_name = session.get('account_name', '未命名')
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        
        # 获取涉及的资产类型
        q1_answer = next((a for a in session['answers'] if a['question_id'] == '1'), None)
        asset_types = ', '.join(q1_answer['structured_answer']) if q1_answer else '未知'
        
        return f"""# CRS信息披露诊断报告

## 一、诊断概要

- 账套名称: {account_name}
- 诊断时间: {timestamp}
- 涉及资产类型: {asset_types}"""
    
    def _generate_asset_overview(self, risk_assessment):
        """生成资产总览表格"""
        assets = risk_assessment['assets']
        
        # 构建表格
        table = """## 二、资产总览

| 资产类型 | 是否涉及 | 风险等级 |
|----------|----------|----------|"""
        
        all_asset_types = ["境外银行存款", "境外证券投资", "离岸公司", "境外保单", "离岸信托"]
        involved_assets = {a['asset_type']: a for a in assets}
        
        risk_emoji = {
            "高风险": "🔴",
            "中风险": "🟡",
            "低风险": "🟢"
        }
        
        for asset_type in all_asset_types:
            if asset_type in involved_assets:
                asset = involved_assets[asset_type]
                risk_level = f"{risk_emoji[asset['risk_level']]} {asset['risk_level']}"
                table += f"\n| {asset_type} | 是 | {risk_level} |"
            else:
                table += f"\n| {asset_type} | 否 | - |"
        
        return table
    
    def _generate_asset_analysis(self, session, risk_assessment):
        """生成各资产详细分析"""
        sections = ["## 三、各资产信息披露义务判定\n"]
        
        for idx, asset in enumerate(risk_assessment['assets'], start=1):
            section = f"### 3.{idx} {asset['asset_type']}\n"
            
            # 基本信息
            section += "\n#### 基本信息\n"
            for key, value in asset.get('basic_info', {}).items():
                section += f"- **{key}**: {value}\n"
            
            # CRS信息披露义务判定
            section += "\n#### CRS信息披露义务判定\n"
            section += f"**判定结果**: {'✅ ' if asset['disclosure_obligation'] == '有披露义务' else '❌ '}{asset['disclosure_obligation']}\n"
            
            section += "\n**理由**:\n"
            for idx_factor, factor in enumerate(asset['risk_factors'], start=1):
                section += f"{idx_factor}. {factor}\n"
            
            section += f"\n综合判定: 该资产属于{asset['risk_level']}类别。\n"
            
            # 信息交换预期(这部分需要AI生成描述性文本)
            section += "\n#### 信息交换预期\n"
            section += "[此处由AI生成描述性文本]\n"
            
            sections.append(section)
        
        return "\n".join(sections)
    
    def _generate_risk_assessment(self, risk_assessment):
        """生成风险评估章节"""
        overall_risk = risk_assessment['overall_risk_level']
        risk_points = risk_assessment['risk_points']
        
        section = f"""## 四、CRS信息交换风险评估

### 4.1 总体风险等级
**{overall_risk}**

[此处由AI生成风险等级描述]

### 4.2 关键风险点
"""
        
        for idx, risk_point in enumerate(risk_points, start=1):
            section += f"\n{idx}. **{risk_point['title']}**\n"
            section += f"   \n   {risk_point['description']}\n"
            section += f"   \n   **影响**: {risk_point['impact']}\n"
            section += f"   \n   **紧急程度**: {risk_point['urgency']}\n"
        
        return section
    
    def _generate_compliance_advice(self, risk_assessment):
        """生成合规建议"""
        overall_risk = risk_assessment['overall_risk_level']
        
        # 根据风险等级生成不同的建议结构
        if overall_risk == "高风险":
            return """## 五、合规建议

### 立即行动(1个月内)
[此处由AI生成具体建议]

### 短期行动(3-6个月内)
[此处由AI生成具体建议]

### 长期规划(6-12个月)
[此处由AI生成具体建议]"""
        
        elif overall_risk == "中风险":
            return """## 五、合规建议

### 短期行动(3-6个月内)
[此处由AI生成具体建议]

### 长期规划(6-12个月)
[此处由AI生成具体建议]

### 日常维护
[此处由AI生成具体建议]"""
        
        else:
            return """## 五、合规建议

### 保持合规
[此处由AI生成具体建议]

### 预防性措施
[此处由AI生成具体建议]"""
    
    def _generate_references(self):
        """生成参考依据(固定内容)"""
        return """## 六、参考依据

- OECD《共同申报准则》(CRS)
- 《非居民金融账户涉税信息尽职调查管理办法》
- 中国签署的税收情报交换协议

---

*本报告由CRS信息披露诊断智能体自动生成,仅供参考,不构成法律意见。建议就具体情况咨询专业税务顾问。*"""
