#### question_flow.py (流程控制脚本)

```python
#!/usr/bin/env python3
"""
问题流程控制脚本
功能: 根据当前问题和用户答案,返回下一个问题
"""

import json
import sys
from pathlib import Path

def load_questions():
    """加载问题清单"""
    questions_file = Path(__file__).parent.parent / 'reference' / 'questions.json'
    with open(questions_file, 'r', encoding='utf-8') as f:
        return json.load(f)

def get_next_question(current_id, user_answer):
    """
    获取下一个问题
    
    Args:
        current_id: 当前问题ID (如: "1-2")
        user_answer: 用户答案 (如: "本人独有")
    
    Returns:
        JSON格式的下一个问题信息
    """
    questions = load_questions()['questions']
    
    # 查找当前问题
    current_question = next((q for q in questions if q['id'] == current_id), None)
    if not current_question:
        return {"error": f"问题 {current_id} 不存在"}
    
    # 根据跳转逻辑确定下一个问题
    jump_logic = current_question.get('jump_logic')
    
    if jump_logic:
        # 有条件跳转
        next_id = jump_logic.get(user_answer)
        if not next_id:
            # 如果用户答案不在跳转逻辑中,使用默认跳转
            next_id = current_question.get('next')
    else:
        # 无条件跳转,使用next字段
        next_id = current_question.get('next')
    
    if not next_id:
        # 当前分类结束
        return {"status": "category_complete", "current_category": current_question['category']}
    
    # 查找下一个问题
    next_question = next((q for q in questions if q['id'] == next_id), None)
    if not next_question:
        return {"error": f"下一个问题 {next_id} 不存在"}
    
    return {
        "status": "continue",
        "next_question": next_question
    }

if __name__ == "__main__":
    if len(sys.argv) != 3:
        print(json.dumps({"error": "用法: python question_flow.py <current_id> <user_answer>"}))
        sys.exit(1)
    
    current_id = sys.argv[1]
    user_answer = sys.argv[2]
    
    result = get_next_question(current_id, user_answer)
    print(json.dumps(result, ensure_ascii=False, indent=2))
```

#### match_jurisdiction.py (地区匹配脚本)

```python
#!/usr/bin/env python3
"""
地区名称匹配脚本
功能: 根据用户输入的地区名称,返回标准地区信息和CRS状态
"""

import json
import sys
from pathlib import Path

def load_jurisdictions():
    """加载CRS参与国名单"""
    jurisdictions_file = Path(__file__).parent.parent / 'reference' / 'crs_jurisdictions.json'
    with open(jurisdictions_file, 'r', encoding='utf-8') as f:
        return json.load(f)['jurisdictions']

def match_jurisdiction(user_input):
    """
    匹配地区名称
    
    Args:
        user_input: 用户输入的地区名称
    
    Returns:
        匹配到的地区信息(JSON)
    """
    jurisdictions = load_jurisdictions()
    user_input_lower = user_input.lower().strip()
    
    # 尝试匹配
    for jurisdiction in jurisdictions:
        # 匹配中文名
        if user_input == jurisdiction['name_cn']:
            return jurisdiction
        
        # 匹配英文名(不区分大小写)
        if user_input_lower == jurisdiction['name_en'].lower():
            return jurisdiction
        
        # 匹配别名
        for alias in jurisdiction.get('aliases', []):
            if user_input_lower == alias.lower():
                return jurisdiction
    
    # 未匹配到
    return {
        "error": "未找到匹配的国家/地区",
        "user_input": user_input,
        "suggestion": "请检查拼写,或使用常见名称如: 香港、新加坡、瑞士、美国等"
    }

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print(json.dumps({"error": "用法: python match_jurisdiction.py <地区名称>"}))
        sys.exit(1)
    
    user_input = sys.argv[1]
    result = match_jurisdiction(user_input)
    print(json.dumps(result, ensure_ascii=False, indent=2))
```

#### convert_currency.py (货币换算脚本)

```python
#!/usr/bin/env python3
"""
货币换算脚本
功能: 将用户输入的金额换算为美元,并归类到三个范围
"""

import json
import sys
import re

# 汇率配置(定期更新)
EXCHANGE_RATES = {
    'USD': 1.0,
    'HKD': 0.128,  # 1港币 = 0.128美元
    'CNY': 0.14,   # 1人民币 = 0.14美元
    'EUR': 1.1,    # 1欧元 = 1.1美元
    'GBP': 1.3,    # 1英镑 = 1.3美元
    'SGD': 0.74,   # 1新币 = 0.74美元
    'JPY': 0.0067, # 1日元 = 0.0067美元
}

# 金额范围定义
RANGES = [
    {"name": "低于100万美元", "min": 0, "max": 1000000},
    {"name": "100万至250万美元", "min": 1000000, "max": 2500000},
    {"name": "超过250万美元", "min": 2500000, "max": float('inf')}
]

def parse_amount(user_input):
    """
    解析用户输入的金额
    
    Args:
        user_input: 如 "300万美元"、"500万港币"、"3 million USD"
    
    Returns:
        {amount: 数值, currency: 货币单位}
    """
    user_input = user_input.strip().upper()
    
    # 提取数字
    numbers = re.findall(r'[\d.]+', user_input)
    if not numbers:
        return None
    
    amount = float(numbers[0])
    
    # 处理"万"单位
    if '万' in user_input or 'W' in user_input:
        amount *= 10000
    
    # 处理"million"单位
    if 'MILLION' in user_input or 'M' in user_input.upper():
        amount *= 1000000
    
    # 识别货币单位
    currency = None
    for curr in EXCHANGE_RATES.keys():
        if curr in user_input:
            currency = curr
            break
    
    # 如果没有明确单位,根据常见表述推测
    if not currency:
        if '美元' in user_input or 'DOLLAR' in user_input:
            currency = 'USD'
        elif '港币' in user_input or '港元' in user_input:
            currency = 'HKD'
        elif '人民币' in user_input or '元' in user_input:
            currency = 'CNY'
        else:
            currency = 'USD'  # 默认美元
    
    return {"amount": amount, "currency": currency}

def convert_to_usd(amount, currency):
    """换算为美元"""
    rate = EXCHANGE_RATES.get(currency, 1.0)
    return amount * rate

def get_range(amount_usd):
    """归类到范围"""
    for range_info in RANGES:
        if range_info['min'] <= amount_usd < range_info['max']:
            return range_info['name']
    return "未知范围"

def convert_currency(user_input):
    """
    主函数: 货币换算
    """
    parsed = parse_amount(user_input)
    if not parsed:
        return {
            "error": "无法解析金额",
            "user_input": user_input,
            "suggestion": "请提供金额和单位,如: 300万美元、500万港币"
        }
    
    amount_usd = convert_to_usd(parsed['amount'], parsed['currency'])
    range_name = get_range(amount_usd)
    
    return {
        "original_amount": parsed['amount'],
        "original_currency": parsed['currency'],
        "amount_usd": round(amount_usd, 2),
        "range": range_name
    }

if __name__ == "__main__":
    if len(sys.argv) != 2:
        print(json.dumps({"error": "用法: python convert_currency.py <金额表述>"}))
        sys.exit(1)
    
    user_input = sys.argv[1]
    result = convert_currency(user_input)
    print(json.dumps(result, ensure_ascii=False, indent=2))
```

### 3.4 会话状态管理

**实现方式**: 代码脚本持久化存储

#### session_manager.py (会话管理脚本)

```python
#!/usr/bin/env python3
"""
会话状态管理脚本
功能: 保存和读取会话状态,支持中断恢复
"""

import json
import sys
from pathlib import Path
from datetime import datetime

SESSIONS_DIR = Path(__file__).parent.parent / 'sessions'
SESSIONS_DIR.mkdir(exist_ok=True)

def save_answer(session_id, question_id, answer_raw, answer_structured):
    """
    保存答案
    
    Args:
        session_id: 会话ID
        question_id: 问题ID
        answer_raw: 用户原始回答
        answer_structured: 提取的结构化答案(JSON字符串)
    """
    session_file = SESSIONS_DIR / f"{session_id}.json"
    
    # 加载现有会话
    if session_file.exists():
        with open(session_file, 'r', encoding='utf-8') as f:
            session = json.load(f)
    else:
        session = {
            "session_id": session_id,
            "created_at": datetime.now().isoformat(),
            "answers": []
        }
    
    # 添加答案
    session["answers"].append({
        "question_id": question_id,
        "answer_raw": answer_raw,
        "answer_structured": json.loads(answer_structured) if answer_structured else None,
        "timestamp": datetime.now().isoformat()
    })
    
    session["updated_at"] = datetime.now().isoformat()
    
    # 保存
    with open(session_file, 'w', encoding='utf-8') as f:
        json.dump(session, f, ensure_ascii=False, indent=2)
    
    return {"status": "saved", "session_id": session_id}

def load_session(session_id):
    """
    加载会话
    """
    session_file = SESSIONS_DIR / f"{session_id}.json"
    
    if not session_file.exists():
        return {"error": "会话不存在"}
    
    with open(session_file, 'r', encoding='utf-8') as f:
        return json.load(f)

def list_sessions():
    """
    列出所有会话
    """
    sessions = []
    for session_file in SESSIONS_DIR.glob("*.json"):
        with open(session_file, 'r', encoding='utf-8') as f:
            session = json.load(f)
            sessions.append({
                "session_id": session['session_id'],
                "created_at": session['created_at'],
                "updated_at": session.get('updated_at'),
                "answer_count": len(session['answers'])
            })
    return sessions

if __name__ == "__main__":
    command = sys.argv[1] if len(sys.argv) > 1 else None
    
    if command == "save":
        # python session_manager.py save <session_id> <question_id> <answer_raw> [answer_structured]
        session_id = sys.argv[2]
        question_id = sys.argv[3]
        answer_raw = sys.argv[4]
        answer_structured = sys.argv[5] if len(sys.argv) > 5 else None
        result = save_answer(session_id, question_id, answer_raw, answer_structured)
    
    elif command == "load":
        # python session_manager.py load <session_id>
        session_id = sys.argv[2]
        result = load_session(session_id)
    
    elif command == "list":
        # python session_manager.py list
        result = {"sessions": list_sessions()}
    
    else:
        result = {"error": "用法: python session_manager.py <save|load|list> [参数]"}
    
    print(json.dumps(result, ensure_ascii=False, indent=2))
```
